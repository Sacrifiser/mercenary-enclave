namespace = mercMenu

# on_custom_diplomacy
# Initial event, mainly to save the event target we use throughout
country_event = {
	id = mercMenu.100
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		NOT = { has_country_flag = "fmerc_diplomacy_engaged" }
		FROM = { is_country_type = "fmerc_enclave" }
	}
	immediate = {
		fmerc_reset_diplo_screen = yes
		set_country_flag = "fmerc_diplomacy_engaged"
		FROM = { save_event_target_as = traders }

		if = { limit = { has_country_flag = "fmerc_active_operation" }
			country_event = { id = mercMenu.102 }
		else = {
			country_event = { id = mercMenu.101 }
		} }
	}
}

# Main menu
# We always invoke fmerc_reset_diplo_screen before this screen
country_event = {
	id = mercMenu.101
	title = fmerc_enclave.name
	desc = {
		trigger = {
			text = fmerc_enclave.100b.desc
			text = newline
			success_text = {
				text = "fmerc_enclave.100b.minerals"
				check_variable = { which = "MercDeposit" value > 0 }
			}
			success_text = {
				text = "fmerc_enclave.100b.energy"
				check_variable = { which = "MercDepositEnergy" value > 0 }
			}
			success_text = {
				text = "fmerc_enclave.100b.fee"
				OR = {
					check_variable = { which = "MercDeposit" value > 0 }
					check_variable = { which = "MercDepositEnergy" value > 0 }
				}
			}
		}
	}

	is_triggered_only = yes
	diplomatic = yes
	custom_gui = "enclave_trader_window"
	custom_gui_option = "enclave_trader_option"
	force_open = yes

	picture_event_data = {
		portrait = event_target:traders
		room = enclave_trader_room
	}

	immediate = {
		fmerc_reset_diplo_screen = yes
		set_country_flag = "fmerc_diplomacy_engaged"
	}

	option = {
		name = fmerc_enclave.100.operations.button
		hidden_effect = { country_event = { id = mercMenu.200 } }
	}
	option = { # Deposit Minerals
		name = fmerc_enclave.100.minerals.button
		custom_tooltip = fmerc_enclave.100.minerals.tooltip
		allow = { minerals > 2500 }
		add_minerals = -2500
		event_target:traders = {
			add_trust = {
				who = ROOT
				amount = 5
			}
		}
		hidden_effect = {
			change_variable = { which = "MercDeposit" value = 2500 }
		}
		is_dialog_only = yes
		response_text = fmerc_enclave.100.minerals.reply
	}
	option = { # Withdraw Minerals
		name = fmerc_enclave.100.minerals.withdraw
		custom_tooltip = fmerc_enclave.100.minerals.withdraw.tooltip
		allow = {
			custom_tooltip_fail = {
				text = "fmerc_enclave.100.minerals.withdraw.fail"
				check_variable = { which = "MercDeposit" value > 2500 }
			}
		}
		add_minerals = 2500
		hidden_effect = {
			subtract_variable = { which = "MercDeposit" value = 2500 }
		}
		is_dialog_only = yes
		response_text = fmerc_enclave.100.minerals.withdraw.reply
	}
	option = { # Deposit Energy
		name = fmerc_enclave.100.energy.button
		custom_tooltip = fmerc_enclave.100.energy.tooltip
		allow = { energy > 2500 }
		add_energy = -2500
		event_target:traders = {
			add_trust = {
				who = ROOT
				amount = 5
			}
		}
		hidden_effect = {
			change_variable = { which = "MercDepositEnergy" value = 2500 }
		}
		is_dialog_only = yes
		response_text = fmerc_enclave.100.energy.reply
	}
	option = { # Withdraw Energy
		name = fmerc_enclave.100.energy.withdraw
		custom_tooltip = fmerc_enclave.100.energy.withdraw.tooltip
		allow = {
			custom_tooltip_fail = {
				text = "fmerc_enclave.100.energy.withdraw.fail"
				check_variable = { which = "MercDepositEnergy" value > 2500 }
			}
		}
		add_energy = 2500
		hidden_effect = {
			subtract_variable = { which = "MercDepositEnergy" value = 2500 }
		}
		is_dialog_only = yes
		response_text = fmerc_enclave.100.energy.withdraw.reply
	}
	option = {
		name = fmerc_enclave.100.stations.button
		custom_tooltip = "fmerc_enclave.100.stations.tooltip"
		trigger = { NOT = { has_country_flag = "fmerc_allow_stations" } }
		hidden_effect = {
			set_country_flag = "fmerc_allow_stations"
			country_event = { id = mercMenu.110 }
		}
	}
	option = {
		name = fmerc_enclave.100.nostations.button
		custom_tooltip = "fmerc_enclave.100.nostations.tooltip"
		trigger = { has_country_flag = "fmerc_allow_stations" }
		allow = {
			custom_tooltip_fail = {
				text = "merc.stationoffer.cooldown"
				NOT = { has_country_flag = "merc_cant_restrict_access" }
			}
		}
		hidden_effect = {
			remove_country_flag = "fmerc_allow_stations"
			country_event = { id = mercMenu.110 }
		}
	}
	option = {
		name = fmerc_enclave.100.insurance.button
		custom_tooltip = "fmerc_enclave.100.insurance.tooltip"
		trigger = { NOT = { has_modifier = merc_enclave_insurance } }
		allow = {
			custom_tooltip_fail = {
				text = fmerc_enclave.has_station
				has_country_flag = "fmerc_allow_stations"
			}
		}
		add_modifier = {
			modifier = merc_enclave_insurance
			days = -1
		}
		hidden_effect = { country_event = { id = mercMenu.150 } }
	}
	option = {
		name = fmerc_enclave.100.insurance.sell
		custom_tooltip = "fmerc_enclave.100.insurance.tooltip"
		trigger = { has_modifier = merc_enclave_insurance }
		remove_modifier = merc_enclave_insurance
		hidden_effect = { country_event = { id = mercMenu.150 } }
	}
	option = {
		name = fmerc_enclave.close
		default_hide_option = yes
		hidden_effect = { remove_country_flag = "fmerc_diplomacy_engaged" }
	}
}
