# on yearly pulse to deduct deposit

event = {
	id = mercRecurring.100
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		every_country = {
			limit = {
				OR = {
					check_variable = { which = "MercDeposit" value > 0 }

			}
			multiply_variable = {
				which = "MercDeposit"
				value = 0.95
			}
			multiply_variable = {
				which = "MercDepositEnergy"
				value = 0.95
			}
		}
	}
}
