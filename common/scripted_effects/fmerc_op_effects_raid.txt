# These effects are only used by the raid operation types

# THIS is op country
merc_set_target_military = {
	event_target:merc_country = {
		random_owned_fleet = {
			limit = {
				fleet_size > 8
				fleet_size < 30
				is_ship_class = shipclass_military
				exists = solar_system
			}
			random_owned_ship = {
				log = "Military target [prev.GetName] (P1)."
				merc_set_found_target = yes
			}
		}
		if = { limit = { PREV = { NOT = { has_country_flag = "merc_found_target" } } }
			random_owned_fleet = {
				limit = {
					is_ship_class = shipclass_military
					fleet_size < 100
					exists = solar_system
				}
				random_owned_ship = {
					log = "Military target [prev.GetName] (P2)."
					merc_set_found_target = yes
				}
			}
		}
		# XXX add more stages maybe?
		if = { limit = { PREV = { NOT = { has_country_flag = "merc_found_target" } } }
			random_owned_fleet = {
				limit = { is_ship_class = shipclass_military }
				random_owned_ship = {
					log = "Military target [prev.GetName] (P3)."
					merc_set_found_target = yes
				}
			}
		}
	}
}

# THIS is op country
merc_set_target_civilian = {
	event_target:merc_country = {
		# First we try to find any targets that are "perfect".
		# That means:
		# 1. No military presence in system
		# 2. No spaceports
		# 3. No colonies in system
		random_owned_fleet = {
			limit = { merc_raid_priority_1 = yes }
			random_owned_ship = {
				log = "Civilian target [prev.GetName] (P1)."
				merc_set_found_target = yes
			}
		}

		if = { limit = { PREV = { NOT = { has_country_flag = "merc_found_target" } } }
			# As a secondary priority, if we can't find only perfect targets
			# - Strike the "no colony" requirement
			random_owned_fleet = {
				limit = { merc_raid_priority_2 = yes }
				random_owned_ship = {
					log = "Civilian target [prev.GetName] (P2)."
					merc_set_found_target = yes
				}
			}
		}

		if = { limit = { PREV = { NOT = { has_country_flag = "merc_found_target" } } }
			# Tertiary priority
			# - Strike the "no spaceports" requirement
			random_owned_fleet = {
				limit = { merc_raid_priority_3 = yes }
				random_owned_ship = {
					log = "Civilian target [prev.GetName] (P3)."
					merc_set_found_target = yes
				}
			}
		}

		if = { limit = { PREV = { NOT = { has_country_flag = "merc_found_target" } } }
			# Finally
			# - Strike the "no military stations" requirement
			random_owned_fleet = {
				limit = { merc_raid_priority_minimum = yes }
				random_owned_ship = {
					log = "Civilian target [prev.GetName] (P4)."
					merc_set_found_target = yes
				}
			}
		}
	}
}
