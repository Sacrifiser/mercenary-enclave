
# Effects for space clearing operations

fmerc_stage2_continue_clear = {
	if = {
		limit = { exists = event_target:fmerc_target }
		set_relation_flag = {
			who = event_target:fmerc_target_country
			flag = merc_operations_target
		}
		else = {
			set_country_flag = "fmerc_ops_completed"
		}
	}
}

fmerc_stage2_init_amoeba = {
	event_target:fmerc_buyer = {
		every_planet_within_border = {
			limit = {
				is_star = yes
				exists = solar_system
				solar_system = {
					any_ship_in_system = {
						exists = owner
						owner = {
							is_country_type = amoeba
						}
					}
				}
			}
			solar_system = {
				random_fleet_in_system = {
					limit = {
						exists = owner
						owner = {
							is_country_type = amoeba
						}
					}
					owner = {
						save_event_target_as = fmerc_target_country
					}
					save_event_target_as = fmerc_target
				}
			}
		}
	}
	fmerc_stage2_continue_clear = yes
}
