merc_has_presence = {
	any_ship_in_system = {
		exists = owner
		owner = {
			OR = {
				is_country_type = "fmerc_enclave"
				is_country_type = "fmerc_operations"
			}
		}
	}
}

merc_has_station = {
	any_ship_in_system = {
		exists = owner
		owner = { is_country_type = "fmerc_enclave" }
		NOT = { is_ship_size = science }
		NOT = { is_ship_size = constructor }
		is_in_combat = no
	}
}

merc_is_station = {
	exists = owner
	owner = { is_country_type = "fmerc_enclave" }
	NOT = { is_ship_size = science }
	NOT = { is_ship_size = constructor }
}

merc_has_contact = {
	any_country = {
		is_country_type = "fmerc_enclave"
		OR = {
			has_communications = PREV
			reverse_has_relation_flag = {
				flag = ongoing_merc_enclave_investigation
				who = PREV
			}
		}
	}
}

merc_needs_contact = {
	exists = capital_scope
	NOT = { is_country_type = "fmerc_operations" }
	NOT = { is_country_type = "fmerc_enclave" }
	num_owned_planets > 0
	merc_has_contact = no
}

fmerc_enclave_exists = {
	any_country = {
		is_country_type = "fmerc_enclave"
		num_ships > 0
	}
}

###############################################################################
# Civilian raid triggers
# All these triggers are limiting a random_owned_fleet scope.
#

merc_raid_priority_1 = {
	merc_raid_priority_minimum = yes
	solar_system = {
		# Priority 1
		NOT = { any_planet = { is_colony = yes } }
		NOT = {
			any_ship_in_system = {
				OR = {
					is_ship_class = shipclass_military_station
					is_ship_class = shipclass_military
					is_ship_class = shipclass_orbital_station # Spaceport
				}
			}
		}
	}
}

merc_raid_priority_2 = {
	merc_raid_priority_minimum = yes
	solar_system = {
		# Priority 2
		NOT = {
			any_ship_in_system = {
				OR = {
					is_ship_class = shipclass_military_station
					is_ship_class = shipclass_military
					is_ship_class = shipclass_orbital_station # Spaceport
				}
			}
		}
	}
}

merc_raid_priority_3 = {
	merc_raid_priority_minimum = yes
	solar_system = {
		# Priority 3
		NOT = {
			any_ship_in_system = {
				OR = {
					is_ship_class = shipclass_military_station
					is_ship_class = shipclass_military
				}
			}
		}
	}
}

merc_raid_priority_minimum = {
	# Lowest denominator
	exists = solar_system
	NOT = { any_owned_ship = { has_ship_flag = "fmerc_target" } }
	merc_valid_civilian_raid_target = yes
	solar_system = { merc_has_presence = no }
}
