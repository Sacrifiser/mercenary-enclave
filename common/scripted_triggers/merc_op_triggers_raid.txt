# Civilian raid triggers
# All these triggers are limiting a random_owned_fleet scope.
#
# XXX should probably use these same ones for the other raid type operations
# XXX as well, because the mercenaries dont actually want to fight more than
# XXX they have to.
#

merc_raid_priority_1 = {
	merc_raid_priority_minimum = yes
	solar_system = {
		# Priority 1
		NOT = { any_planet = { is_colony = yes } }
		NOT = {
			any_ship_in_system = {
				OR = {
					is_ship_class = shipclass_military_station
					is_ship_class = shipclass_military
					is_ship_class = shipclass_orbital_station # Spaceport
				}
			}
		}
	}
}

merc_raid_priority_2 = {
	merc_raid_priority_minimum = yes
	solar_system = {
		# Priority 2
		NOT = {
			any_ship_in_system = {
				OR = {
					is_ship_class = shipclass_military_station
					is_ship_class = shipclass_military
					is_ship_class = shipclass_orbital_station # Spaceport
				}
			}
		}
	}
}

merc_raid_priority_3 = {
	merc_raid_priority_minimum = yes
	solar_system = {
		# Priority 3
		NOT = {
			any_ship_in_system = {
				OR = {
					is_ship_class = shipclass_military_station
					is_ship_class = shipclass_military
				}
			}
		}
	}
}

merc_raid_priority_minimum = {
	# Lowest denominator
	exists = solar_system
	merc_valid_civilian_raid_target = yes
	solar_system = { merc_has_station = no }
}
