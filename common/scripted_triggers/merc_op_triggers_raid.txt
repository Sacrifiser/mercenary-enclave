merc_raid_p1 = {
	merc_raid_minimum = yes
	solar_system = { NOT = { any_planet = { is_colony = yes } } }
}

	merc_raid_minimum = yes
	solar_system = { NOT = { any_planet = { is_capital = yes } } }
}

merc_raid_minimum = {
	exists = solar_system
	solar_system = { merc_has_station = no }
}
